async function registrar() { const btn = document.getElementById('btn-registrar'); const err = document.getElementById('error-paso3'); const user = document.getElementById('reg-username').value.trim(); const nom = document.getElementById('reg-nombre').value.trim(); const bio = document.getElementById('reg-bio').value.trim(); const nombreLegal = document.getElementById('reg-nombre-legal').value.trim(); if (!user || !nom) return mostrarError(err, 'Username y nombre son obligatorios.'); btn.disabled = true; btn.textContent = 'Procesando...'; const { error } = await client.auth.signUp({ email: document.getElementById('reg-email').value, password: document.getElementById('reg-password').value, options: { data: { username: user, nombre_display: nom, tipo: tipoSeleccionado } } }); if (error) { btn.disabled = false; btn.textContent = 'Crear cuenta'; return mostrarError(err, error.message); } // Esperamos un momento para que Supabase cree el usuario en la tabla 'perfiles' con sus triggers setTimeout(async () => { const { data: { user } } = await client.auth.getUser(); if (user) { // Inyectamos la estructura vacía desde el día 1 para evitar errores await client.from('perfiles').update({ bio: bio, generos: generosSeleccionados, nombre_legal: nombreLegal || null, estudios: [], equipamiento: [], links_custom: [], color_acento: '#7c3aed', fuente_nombre: 'Inter' }).eq('id', user.id); } }, 1500); mostrarPaso('exito'); }
